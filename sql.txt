-- 1. Create a table for User Profiles
CREATE TABLE profiles (
  id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
  email TEXT UNIQUE,
  role TEXT CHECK (role IN ('dev', 'admin', 'client')) DEFAULT 'client'
);

-- 2. Create a table for Businesses
CREATE TABLE businesses (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  admin_id UUID REFERENCES profiles(id),
  name TEXT NOT NULL,
  logo_url TEXT,
  address TEXT,
  reg_no TEXT,
  whatsapp TEXT,
  contact_person TEXT,
  business_type TEXT CHECK (business_type IN ('service', 'product', 'both')),
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- 3. Trigger to create a profile record when a new user signs up
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
  INSERT INTO public.profiles (id, email, role)
  VALUES (new.id, new.email, 'admin'); -- Default for auto-signed up users
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 1. Create the Main Products Table
CREATE TABLE public.products (
    id uuid NOT NULL DEFAULT gen_random_uuid(),
    business_id uuid NOT NULL,
    name text NOT NULL,
    description text,
    unit_price numeric DEFAULT 0,
    available_stock integer DEFAULT 0,
    stock_status text DEFAULT 'in_stock'::text,
    remarks text,
    images text[] DEFAULT '{}'::text[], -- Array for up to 5 URLs
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()),
    
    CONSTRAINT products_pkey PRIMARY KEY (id),
    CONSTRAINT products_business_id_fkey FOREIGN KEY (business_id) REFERENCES public.businesses(id) ON DELETE CASCADE
);

-- 2. Create the Variations Table (Name only, relational for filtering)
CREATE TABLE public.product_variations (
    id uuid NOT NULL DEFAULT gen_random_uuid(),
    product_id uuid NOT NULL,
    variation_name text NOT NULL, -- e.g., 'Red', 'XL', '1TB'
    
    CONSTRAINT product_variations_pkey PRIMARY KEY (id),
    CONSTRAINT product_variations_product_id_fkey FOREIGN KEY (product_id) REFERENCES public.products(id) ON DELETE CASCADE
);

-- 3. Enable Row Level Security (RLS)
ALTER TABLE public.products ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.product_variations ENABLE ROW LEVEL SECURITY;

-- 4. Create Access Policies (Admins only see their own business data)
CREATE POLICY "Admins can manage their own products" 
ON public.products 
FOR ALL 
USING (
  business_id IN (
    SELECT id FROM businesses WHERE admin_id = auth.uid()
  )
);

CREATE POLICY "Admins can manage their own variations" 
ON public.product_variations 
FOR ALL 
USING (
  product_id IN (
    SELECT id FROM products WHERE business_id IN (
      SELECT id FROM businesses WHERE admin_id = auth.uid()
    )
  )
);

-- 1. Regular Operating Hours (Weekly)
CREATE TABLE public.business_hours (
    id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
    business_id uuid REFERENCES public.businesses(id) ON DELETE CASCADE,
    day_of_week int NOT NULL, -- 0 (Sunday) to 6 (Saturday)
    open_time time NOT NULL,
    close_time time NOT NULL,
    UNIQUE(business_id, day_of_week, open_time)
);

-- 2. Exceptions (Specific Dates)
CREATE TABLE public.business_exceptions (
    id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
    business_id uuid REFERENCES public.businesses(id) ON DELETE CASCADE,
    exception_date date NOT NULL,
    is_closed_all_day boolean DEFAULT false,
    closed_from time,
    closed_until time,
    reason text
);

-- 3. Auto-Cleanup Function (Deletes past exceptions)
CREATE OR REPLACE FUNCTION delete_past_exceptions() 
RETURNS void AS $$
BEGIN
    DELETE FROM public.business_exceptions WHERE exception_date < CURRENT_DATE;
END;
$$ LANGUAGE plpgsql;
